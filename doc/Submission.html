class Submission < ApplicationRecord
  belongs_to :user
  belongs_to :evaluation

  has_many :submission_answers, dependent: :destroy

  enum state: {
    open: 0,
    submitted: 1,
    graded: 2,
    closed: 3,
    rejected: 4
  }

  validates :state, presence: true
  validates :score, numericality: { greater_than_or_equal_to: 0 }, allow_nil: true

  end

  end
